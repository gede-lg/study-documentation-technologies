class LoginNavigator extends StatelessWidget {
  const LoginNavigator({super.key});

  @override
  Widget build(BuildContext context) {
    return Navigator(
      initialRoute: '/login', // Define 'login' como a rota inicial para este Navigator
      onGenerateRoute: (RouteSettings settings) {
        final Uri uri = Uri.parse(settings.name ?? '');
        switch (settings.name) {
          case '/login':
            return MaterialPageRoute(builder: (_) => const LoginPage());
          case '/login/reset-pwd':
            return MaterialPageRoute(
              settings: settings,
              builder: (_) => const ResetPasswordNavigator(),
            );
          default:
            return MaterialPageRoute(builder: (_) => const LoginPage());
        }
      },
    );
  }
}

class ResetPasswordNavigator extends StatelessWidget {
  const ResetPasswordNavigator({super.key});

  @override
  Widget build(BuildContext context) {
    return Navigator(
      initialRoute: 'welcome',
      onGenerateRoute: (RouteSettings settings) {
        switch (settings.name) {
          case 'verification':
            return MaterialPageRoute(builder: (_) => const VerificationCodePage());
          case 'new-password':
            return MaterialPageRoute(builder: (_) => const UpdatePasswordPage());
          case 'welcome':
          default:
            return MaterialPageRoute(builder: (_) => const WelcomePage());
        }
      },
    );
  }
}
